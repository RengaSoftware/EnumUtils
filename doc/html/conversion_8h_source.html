39  template <typename E, typename T>
40  auto make_reverse_lookup_table()
41  {
42  auto result = std::unordered_map<T, E>{};
43 
44  for (const auto e : range<E>{})
45  {
46  const auto is_duplicate = !result.try_emplace(convert_to_value<T>(e), e).second;
47 
48  if (is_duplicate)
49  throw bad_conversion{"Ambiguous reverse conversion (multiple enum entries match the same value)"};
50  }
51 
52  return result;
53  }

100  template <typename E, typename T>
101  E convert_to_enum(const T& t)
102  {
103  static const auto reverse_lookup_table = detail::make_reverse_lookup_table<E, T>();
104 
105  const auto it = reverse_lookup_table.find(t);
106 
107  if (it == reverse_lookup_table.cend())
108  throw bad_conversion{"No enum value matches the key"};
109 
110  return it->second;
111  }
